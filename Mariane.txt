def main():
  contador = 3
  x = 0
  y = 1
  z = 1

  n = int(input("Digite um número inteiro maior ou igual a ZERO: "))

  if n == 0:
    print("T(",n,") é 0")

  elif n == 1 or n == 2:
    print("T(",n,") é 1")
    
  elif n <= 0:
    print("Número Inválido")
    
  else:
    while contador <= n:
      T = x + y + z
      x = y
      y = z
      z = T
      contador += 1
    print("T(",n,") é ",T,"")

main()
